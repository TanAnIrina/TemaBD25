CREATE TABLE Utilizator (
    ID_Utilizator NUMBER(9) PRIMARY KEY,
    Nume_Utilizator VARCHAR2(20) UNIQUE NOT NULL,
    Email VARCHAR2(50) UNIQUE NOT NULL
);

CREATE TABLE Prietenie (
    ID_Utilizator1 NUMBER(9),
    ID_Utilizator2 NUMBER(9),
    PRIMARY KEY (ID_Utilizator1, ID_Utilizator2),
    FOREIGN KEY (ID_Utilizator1) REFERENCES Utilizator(ID_Utilizator),
    FOREIGN KEY (ID_Utilizator2) REFERENCES Utilizator(ID_Utilizator)
);

CREATE TABLE Club_de_lectura (
    ID_Club NUMBER(9) PRIMARY KEY,
    Nume VARCHAR(50) NOT NULL
);

CREATE TABLE Participare_club (
    ID_Utilizator NUMBER(9),
    ID_Club NUMBER(9),
    PRIMARY KEY (ID_Utilizator, ID_Club),
    FOREIGN KEY (ID_Utilizator) REFERENCES UTILIZATOR(ID_Utilizator),
    FOREIGN KEY (ID_Club) REFERENCES CLUB_DE_LECTURA(ID_Club)
);

CREATE TABLE Concurs_de_lectura (
    ID_Concurs NUMBER(9) PRIMARY KEY,
    Data_Incepere DATE NOT NULL,
    Data_Finalizare DATE NOT NULL
);

CREATE TABLE Participare_Concurs (
    ID_Concurs NUMBER(9),
    ID_Utilizator NUMBER(9),
    PRIMARY KEY (ID_Concurs, ID_Utilizator),
    FOREIGN KEY (ID_Concurs) REFERENCES Concurs_de_Lectura(ID_Concurs),
    FOREIGN KEY (ID_Utilizator) REFERENCES Utilizator(ID_Utilizator)
);

CREATE TABLE Gen_literar (
    ID_Gen NUMBER(9) PRIMARY KEY,
    Nume_Gen VARCHAR2(50)
);

CREATE TABLE Carte (
    ID_Carte NUMBER(9) PRIMARY KEY,
    Titlu VARCHAR2(100) NOT NULL,
    ID_Gen NUMBER(9),
    FOREIGN KEY (ID_Gen) REFERENCES Gen_literar(ID_Gen)
);

CREATE TABLE Autor (
    ID_Autor NUMBER(9) PRIMARY KEY,
    Pseudonim VARCHAR2(100) NOT NULL
);

CREATE TABLE Intalnire_club (
    ID_Intalnire NUMBER(9) PRIMARY KEY,
    Data_Intalnire DATE NOT NULL,
    ID_Club NUMBER(9),
    ID_Autor NUMBER(9),
    FOREIGN KEY (ID_Club) REFERENCES CLUB_DE_LECTURA(ID_Club),
    FOREIGN KEY (ID_Autor) REFERENCES AUTOR(ID_Autor) ON DELETE SET NULL
);

CREATE TABLE Recenzie (
    ID_Recenzie NUMBER(9) PRIMARY KEY,
    ID_Utilizator NUMBER(9),
    ID_Carte NUMBER(9),
    Continut VARCHAR2(500),
    Data DATE,
    FOREIGN KEY (ID_Utilizator) REFERENCES Utilizator(ID_Utilizator),
    FOREIGN KEY (ID_Carte) REFERENCES Carte(ID_Carte)
);

CREATE TABLE Lista_de_lectura (
    ID_Utilizator NUMBER(9),
    ID_Carte NUMBER(9),
    Status VARCHAR2(25 CHAR) CHECK (Status IN ('citita', 'in curs de citire', 'va fi citita')),
    PRIMARY KEY (ID_Utilizator, ID_Carte),
    FOREIGN KEY (ID_Utilizator) REFERENCES Utilizator(ID_Utilizator),
    FOREIGN KEY (ID_Carte) REFERENCES Carte(ID_Carte)
);

CREATE TABLE Colaborare (
    ID_Carte NUMBER(9),
    ID_Autor NUMBER(9),
    PRIMARY KEY (ID_Carte, ID_Autor),
    FOREIGN KEY (ID_Carte) REFERENCES Carte(ID_Carte),
    FOREIGN KEY (ID_Autor) REFERENCES Autor(ID_Autor)
);
